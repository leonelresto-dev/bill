  try { localStorage.setItem(KEY, JSON.stringify(data)); } catch {}
}

function getData() { return load(); }

function getBills() {
  const d = getData();
  if (!d.bills) {
    d.bills = [
      { id:1, name:"Renta / Hipoteca",  category:"Vivienda",      amount:"" },
      { id:2, name:"Electricidad",       category:"Servicios",     amount:"" },
      { id:3, name:"Internet",           category:"Servicios",     amount:"" },
      { id:4, name:"Teléfono",           category:"Servicios",     amount:"" },
      { id:5, name:"Netflix",            category:"Suscripciones", amount:"" },
      { id:6, name:"Seguro de Auto",     category:"Seguros",       amount:"" },
    ];
    save(d);
  }
  return d.bills;
}

function getPayments() {
  const d = getData(); return d.payments || {};
}

function saveAll(bills, payments) {
  save({ bills, payments });
}

function monthKey() { return `${curYear}-${curMonth}`; }
function payKey(billId) { return `${monthKey()}-${billId}`; }

// ── RENDER ────────────────────────────────────────────────────────
function render() {
  updateHeader();
  if (activeView === "month") renderMonth();
  else renderYear();
  updateNavTabs();
}

function updateHeader() {
  const isNow = curMonth === today.getMonth() && curYear === today.getFullYear();
  document.getElementById("month-label").innerHTML =
    `${MONTHS[curMonth]} ${curYear}` +
    (isNow ? `<span class="badge-today">Hoy</span>` : "");

  document.getElementById("tab-month").className = "tab-btn" + (activeView==="month" ? " active" : "");
  document.getElementById("tab-year").className  = "tab-btn" + (activeView==="year"  ? " active" : "");
}

function updateNavTabs() {
  document.getElementById("nav-month").className = "nav-tab" + (activeView==="month" ? " active" : "");
  document.getElementById("nav-year").className  = "nav-tab" + (activeView==="year"  ? " active" : "");
}

// ── MONTH VIEW ────────────────────────────────────────────────────
function renderMonth() {
  const bills    = getBills();
  const payments = getPayments();

  const filtered = filterCat === "Todas" ? bills : bills.filter(b => b.category === filterCat);

  function getAmt(b) {
    const p = payments[payKey(b.id)] || {};
    return p.amount !== undefined ? p.amount : b.amount;
  }

  const paid   = filtered.filter(b => (payments[payKey(b.id)] || {}).paid);
  const unpaid = filtered.filter(b => !(payments[payKey(b.id)] || {}).paid);

  const totalExp  = filtered.reduce((s,b) => s + (parseFloat(getAmt(b))||0), 0);
  const totalPaid = paid.reduce((s,b) => s + (parseFloat(getAmt(b))||0), 0);
  const pct       = totalExp > 0 ? (totalPaid / totalExp) * 100 : 0;

  // Install banner check
  const showBanner = !localStorage.getItem("install_dismissed") && !window.navigator.standalone;

  let html = ``;

  // Install banner
  if (showBanner) {
    html += `<div class="install-banner">
      <div>📲</div>
      <div class="install-text">
        <div class="install-title">Instalar como app</div>
        <div class="install-sub">Toca <b>Compartir</b> → <b>"Agregar a inicio"</b> en Safari</div>
      </div>
      <button class="install-dismiss" onclick="dismissInstall()">OK</button>
    </div>`;
  }

  // Stats
  html += `<div class="stats-grid">
    <div class="stat-card">
      <div class="stat-icon">💰</div>
      <div class="stat-value" style="color:#ffffff">$${fmt(totalExp)}</div>
      <div class="stat-label">Total</div>
    </div>
    <div class="stat-card">
      <div class="stat-icon">✅</div>
      <div class="stat-value" style="color:#10b981">$${fmt(totalPaid)}</div>
      <div class="stat-label">Pagado</div>
    </div>
    <div class="stat-card">
      <div class="stat-icon">⏳</div>
      <div class="stat-value" style="color:#f59e0b">$${fmt(totalExp - totalPaid)}</div>
      <div class="stat-label">Pendiente</div>
    </div>
  </div>`;

  // Progress
  html += `<div class="progress-wrap">
    <div class="progress-meta">
      <span class="progress-text">${paid.length} de ${filtered.length} bills pagados</span>
      <span class="progress-pct">${pct.toFixed(0)}%</span>
    </div>
    <div class="progress-bar-bg">
      <div class="progress-bar-fill" style="width:${pct}%"></div>
    </div>
  </div>`;

  // Category chips
  html += `<div class="cat-filter">`;
  CATS.forEach(cat => {
    const c = CAT_COLORS[cat];
    const isActive = filterCat === cat;
    const style = isActive && c ? `background:${c.color};border-color:${c.color};color:#fff` : (isActive ? `background:#ffffff;border-color:#ffffff;color:#000000` : "");
    html += `<button class="cat-chip${isActive?" active":""}" style="${style}" onclick="setFilter('${cat}')">${cat}</button>`;
  });
  html += `</div>`;

  // Unpaid
  if (unpaid.length > 0) {
    html += `<div>
      <div class="section-label" style="color:#f59e0b">⏳ Pendientes <span class="section-count">${unpaid.length}</span></div>
      <div class="bills-list">${unpaid.map(b => billCard(b, payments, getAmt(b))).join("")}</div>
    </div>`;
  }

  // Paid
  if (paid.length > 0) {
    html += `<div>
      <div class="section-label" style="color:#10b981">✅ Pagados <span class="section-count">${paid.length}</span></div>
      <div class="bills-list">${paid.map(b => billCard(b, payments, getAmt(b))).join("")}</div>
    </div>`;
  }

  if (filtered.length === 0) {
    html += `<div class="empty-state">
      <div class="empty-icon">🧾</div>
      <div class="empty-title">Sin bills registrados</div>
      <div class="empty-sub">Toca "+ Nuevo Bill" arriba para agregar tu primer pago</div>
    </div>`;
  }

  document.getElementById("main").innerHTML = html;

  // Re-attach amount input if editing
  if (editingAmountId) {
    const inp = document.getElementById(`amt-inp-${editingAmountId}`);
    if (inp) { inp.focus(); inp.select(); }
  }
}

function billCard(b, payments, amount) {
  const p = payments[payKey(b.id)] || {};
  const isPaid = !!p.paid;
  const c = CAT_COLORS[b.category] || CAT_COLORS["Otros"];
  const isEditing = editingAmountId === b.id;
  const displayAmt = parseFloat(amount || 0).toLocaleString("en-US", { minimumFractionDigits: 0, maximumFractionDigits: 2 });
  const paidDate = p.paidAt ? new Date(p.paidAt).toLocaleDateString("es-MX",{day:"numeric",month:"short"}) : "";

  const amountEl = isEditing
    ? `<input id="amt-inp-${b.id}" class="amount-input" type="number" inputmode="decimal" value="${amount||""}"
         onblur="saveAmount(${b.id}, this.value)"
         onkeydown="if(event.key==='Enter')this.blur()"
         onclick="event.stopPropagation()">`
    : `<div class="bill-amount" style="color:${isPaid?'#10b981':'#ffffff'}" onclick="event.stopPropagation();startEditAmount(${b.id})">$${displayAmt}</div>`;

  return `<div class="bill-card${isPaid?" paid":""}" onclick="togglePaid(${b.id})">
    <button class="check-btn${isPaid?" checked":""}" onclick="event.stopPropagation();togglePaid(${b.id})">${isPaid?"✓":""}</button>
    <div class="bill-info">
      <div class="bill-name">${escHtml(b.name)}</div>
      <div class="bill-meta">
        <span class="bill-cat" style="color:${c.color};background:${c.bg}">${b.category}</span>
        ${paidDate ? `<span class="bill-date">• ${paidDate}</span>` : ""}
      </div>
    </div>
    <div class="bill-right">
      ${amountEl}
      <button class="delete-btn" onclick="event.stopPropagation();deleteBill(${b.id})">✕ Eliminar</button>
    </div>
  </div>`;
}

// ── YEAR VIEW ─────────────────────────────────────────────────────
function renderYear() {
  const bills    = getBills();
  const payments = getPayments();

  let html = `<div class="year-grid">`;

  MONTHS.forEach((mName, idx) => {
    const mk = `${curYear}-${idx}`;
    const isCurrent = idx === today.getMonth() && curYear === today.getFullYear();

    let total = 0, paid = 0, paidCount = 0;
    bills.forEach(b => {
      const p = payments[`${mk}-${b.id}`] || {};
      const a = parseFloat(p.amount !== undefined ? p.amount : b.amount) || 0;
      total += a;
      if (p.paid) { paid += a; paidCount++; }
    });

    const pct = total > 0 ? (paid / total) * 100 : 0;
    const barColor = pct === 100 ? "#10b981" : pct > 0 ? "linear-gradient(90deg,#ffffff,#aaaaaa)" : "#222222";

    html += `<div class="year-card${isCurrent?" current":""}" onclick="setMonth(${idx})">
      <div class="year-month-name">${mName}${isCurrent?` <span class="badge-today">Hoy</span>`:""}</div>
      <div class="year-bar-area">
        <div class="year-bar-bg">
          <div class="year-bar-fill" style="width:${pct}%;background:${barColor}"></div>
        </div>
        <div class="year-meta">
          <span class="year-paid">$${fmt(paid)} pagado</span>
          <span class="year-count">${paidCount}/${bills.length} bills · $${fmt(total)}</span>
        </div>
      </div>
    </div>`;
  });

  html += `</div>`;
  document.getElementById("main").innerHTML = html;
}

// ── ACTIONS ───────────────────────────────────────────────────────
function togglePaid(billId) {
  const bills    = getBills();
  const payments = getPayments();
  const k = payKey(billId);
  const cur = payments[k] || {};
  payments[k] = { ...cur, paid: !cur.paid, paidAt: !cur.paid ? new Date().toISOString() : null };
  saveAll(bills, payments);
  const isPaid = payments[k].paid;
  const b = bills.find(x => x.id === billId);
  if (isPaid && b) toast(`✅ ${b.name} marcado pagado`);
  render();
}

function startEditAmount(billId) {
  editingAmountId = billId;
  render();
}

function saveAmount(billId, value) {
  const bills    = getBills();
  const payments = getPayments();
  const k = payKey(billId);
  payments[k] = { ...(payments[k]||{}), amount: value };
  saveAll(bills, payments);
  editingAmountId = null;
  render();
}

function deleteBill(billId) {
  const d = getData();
  d.bills = (d.bills || []).filter(b => b.id !== billId);
  save(d);
  render();
}

function changeMonth(dir) {
  curMonth += dir;
  if (curMonth < 0) { curMonth = 11; curYear--; }
  if (curMonth > 11) { curMonth = 0; curYear++; }
  render();
}

function setMonth(idx) {
  curMonth = idx;
  activeView = "month";
  render();
}

function setView(v) {
  activeView = v;
  render();
}

function setFilter(cat) {
  filterCat = cat;
  render();
}

function openModal() {
  document.getElementById("modal").style.display = "flex";
  setTimeout(() => document.getElementById("input-name").focus(), 300);
}

function closeModal() {
  document.getElementById("modal").style.display = "none";
  document.getElementById("input-name").value = "";
  document.getElementById("input-amount").value = "";
  document.getElementById("input-cat").value = "Otros";
}

function saveBill() {
  const name   = document.getElementById("input-name").value.trim();
  const cat    = document.getElementById("input-cat").value;
  const amount = document.getElementById("input-amount").value;
  if (!name) { document.getElementById("input-name").focus(); return; }
  const d = getData();
  if (!d.bills) d.bills = getBills();
  const newBill = { id: Date.now(), name, category: cat, amount };
  d.bills.push(newBill);
  save(d);
  closeModal();
  toast(`💳 "${name}" agregado`);
  render();
}

function dismissInstall() {
  localStorage.setItem("install_dismissed","1");
  render();
}

function toast(msg) {
  const el = document.getElementById("toast");
  el.textContent = msg;
  el.classList.add("show");
  clearTimeout(el._t);
  el._t = setTimeout(() => el.classList.remove("show"), 2200);
}

function fmt(n) {
  const v = parseFloat(n) || 0;
  return v.toLocaleString("en-US", { minimumFractionDigits: 0, maximumFractionDigits: 0 });
}

function escHtml(s) {
  return String(s).replace(/&/g,"&amp;").replace(/</g,"&lt;").replace(/>/g,"&gt;").replace(/"/g,"&quot;");
}

// ── KEYBOARD CLOSE ────────────────────────────────────────────────
document.addEventListener("keydown", e => {
  if (e.key === "Escape") {
    closeModal();
    if (editingAmountId) { editingAmountId = null; render(); }
  }
});
